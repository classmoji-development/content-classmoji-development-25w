Common React Gotchas
// BAD: Stale closure
const [count, setCount] = useState(0);

useEffect(() => {
  const interval = setInterval(() => {
    setCount(count + 1); // count is stale!
  }, 1000);
  return () => clearInterval(interval);
}, []); // empty deps - count never updates

// GOOD: Use functional update
setCount(prevCount => prevCount + 1);
Gotcha #1: Stale Closures
// BAD: Creates new object every render = infinite loop!
const [user, setUser] = useState({ name: 'Tim' });

useEffect(() => {
  // This runs every render, causing another render...
  setUser({ name: 'Tim' });
}, [user]); // user is new object each time!

// GOOD: Use primitive values or useMemo
useEffect(() => {
  setUser({ name: 'Tim' });
}, [user.name]); // only re-run if name changes
Gotcha #2: Infinite Render Loops
// BAD: Using array index as key
{items.map((item, index) =>
  <div key={index}>{item.name}</div>
)}

// PROBLEM: If list reorders, React gets confused!
// Items may lose state or not update correctly

// GOOD: Use stable, unique identifiers
{items.map((item) =>
  <div key={item.id}>{item.name}</div>
)}
Gotcha #3: Key Props in Lists
// BAD: Mutating state directly
const [items, setItems] = useState([1, 2, 3]);

items.push(4); // NEVER DO THIS!
setItems(items); // React won't detect change

// GOOD: Create new array/object
setItems([...items, 4]); // spread operator
setItems(items.concat(4)); // or concat
setItems(prevItems => [...prevItems, 4]); // functional update

// For objects:
setUser({ ...user, name: 'New Name' });
Gotcha #4: Direct State Mutation
// BAD: Missing dependencies
const [userId, setUserId] = useState(1);

useEffect(() => {
  fetchUser(userId).then(setUser);
}, []); // userId not in deps! Won't update when userId changes

// GOOD: Include all dependencies
useEffect(() => {
  fetchUser(userId).then(setUser);
}, [userId]); // re-run when userId changes

// PRO TIP: Use ESLint plugin 'eslint-plugin-react-hooks'
// to catch missing dependencies automatically!

Rule of thumb: If you use it inside useEffect, it goes in the dependency array!
