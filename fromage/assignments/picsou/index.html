class Car {
  constructor(make, model, year, color) {
    this.make = make;
    this.model = model;
    this.year = year;
    this.color = color;
    this.isRunning = false;
    this.speed = 0;
  }

  start() {
    this.isRunning = true;
    console.log(`${this.year} ${this.make} ${this.model} is now running.`);
  }

  stop() {
    this.isRunning = false;
    this.speed = 0;
    console.log(`${this.make} ${this.model} has stopped.`);
  }

  accelerate(amount) {
    if (this.isRunning) {
      this.speed += amount;
      console.log(`Speed increased to ${this.speed} mph.`);
    } else {
      console.log('Car is not running. Start the car first.');
    }
  }

  brake(amount) {
    if (this.speed > 0) {
      this.speed = Math.max(0, this.speed - amount);
      console.log(`Speed reduced to ${this.speed} mph.`);
    }
  }

  getInfo() {
    return `${this.year} ${this.color} ${this.make} ${this.model}`;
  }
}

// Usage example
const myCar = new Car('Toyota', 'Camry', 2023, 'blue');
myCar.start();
myCar.accelerate(30);
myCar.brake(10);
myCar.stop();
console.log(myCar.getInfo());
